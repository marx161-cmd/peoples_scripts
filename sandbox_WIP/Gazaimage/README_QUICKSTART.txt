# Quickstart
1. termux-setup-storage
2. pkg install unzip python libxml2 libxslt clang -y
3. python -m venv venv && . venv/bin/activate
4. pip install --upgrade pip
5. pip install requests beautifulsoup4 python-dateutil lxml
6. unzip gaza_scraper_termux_full_v2.zip -d ~/gaza_scraper
7. chmod +x run_crawl.sh
8. ./run_crawl.sh
